SuperPotato Filesystem (spfs)


SPFS_HEADER:
	byte  Reserved[3] //jmp <addresss> nop
	word  Version	  
	word  Reserved
	byte  VolumeLabel[16]
	word  BytesPerSector
	byte  ReservedSectors
	qword NumDataSectors
	qword NumAllocationTables
	qword LBA

SPFS_DATE_TIME:
	word time //bits 0-5 minutes, 6-10 hours, 11-15 days
	word date // bits 0-11 year (year between 2000-3000), 12-15 month

SPFS_FILE_ENTRY
	byte  			Type
	byte 			Reserved
	byte			Reserved
	byte  			Name[40]
	SPFS_DATE_TIME	creationDate
	qword 			Size
	qword 			LBA
